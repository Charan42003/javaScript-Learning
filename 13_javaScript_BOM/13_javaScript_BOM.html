Browser Object Model - BOM

window object 

        window.innerHeight; // browser display height
        window.innerWidth;  // browser display width
        window.OuterWidth;  // full browser width
        window.OuterHeight; // full browser height

Action 

    to open:
        element.addEventListner('click') function() {
            window.open(url, name, features);
        }

        //there are more features to use

        example:
            let link = document.querySelector('#button');
            let url = "https://www.google.com";
            let name = "google"
            let features = "height=500, width=300";
            link.addEventListener('click', function() {
                window.open(url, google, features);
            });

    to close:
        element.addEventListner('click') function() {
            window.close(window to close);
        }

        example:
            link.addEventListener('click', function() {
                win = window.open(url, name, features);
            });
            let link1 = document.querySelector('#button1');
            link1.addEventListener('click', function() {
                window.close(win);
            });

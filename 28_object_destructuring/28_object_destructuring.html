Object Destructuring

allows you to neatly extract an object value into new variables
note: variable name and property name should be same 

        syntax:
            let{property_name_1, property_name_2, ..., property_name_n} = object;
                for nested array
            let{property_name_1, property_name_2, ..., nested_Object : {nest_property_1, ..., nest_property_n}} = object;

            objects can be directly written in the place of 'object'

        example:
            let data = {
                name: "charan",
                age: 19,
                obj: {
                    college: "ABCD",
                    place: "Bangalore",
                    pin: 460049
                }
            };
            
            let{name : n, age, obj : {college, place, pin}} = data;
            console.log(n);
            console.log(age);
            console.log(college);
            console.log(place);
            console.log(pin);

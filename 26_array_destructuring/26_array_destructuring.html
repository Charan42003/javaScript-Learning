Array Destructuring

allows you to neatly extract an array's value into new variables

        syntax:
            let[name_1, name_2, ..., name_n] = [array_value1, array_value2, ..., array_valuen];
                for nested array
            let[name_1, name_2, [nest1, nest2]] = [array_value1, array_value2, [nestValue1, nestValue2]];

        example:
            let[namee, page, price] = ["Advance JS", 200, 3000];
            console.log(namee);
            console.log(page);
            console.log(price);

            let[a1, a2, [b1, b2]] = ["a_value1", "a_value2", ["b_value1", "b_value2"]];
            console.log(a1 + " " + a2 + " " + b1 + " " + b2)
        
        usecase:
            function rand() {
                return [true, false, "NaN"];
            }
            let[t, f, n] = rand();
            console.log(t + " " + f + " " + n);

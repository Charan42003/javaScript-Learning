redirect (setlocation path)

    1.
        window.location = 'path';

        example:
            window.location = "https://www.yahoo.com";
            
    2.
        location.href = 'path';

        example:
           location.href = "https://www.yahoo.com";

    3.
        location.assign('path');

        example:
            location.assign("https://www.instagram.com");
    
    4. 
        location.replace('path');

        example:
            location.replace("https://www.instagram.com");

    5.
        location.reload();

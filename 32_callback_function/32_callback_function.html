Callback Function

can pass a function into another function as a parameter 

        function sayHello(n) {
            console.log("hello "+n)
        }
         
        function process(b, func) {
            func(b);
        }
        
        process("charan", sayHello)

        // parameter can also be an ananomous function 
        function proc(a, b) {
            console.log(a);
            b();
        }
        proc(5, function() {
            console.log("this is an ananomous function")
        })
